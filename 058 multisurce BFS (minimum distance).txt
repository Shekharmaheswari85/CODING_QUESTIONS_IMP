#include<bits/stdc++.h>
using namespace std;
int main() {
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    int n,m;
    cin >> n>>m;
    int a[n][m], dist[n][m];
    memset(dist,0,sizeof(dist));
    queue<pair<int,int>> q;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> a[i][j];
            dist[i][j]=INT_MAX;
            if(a[i][j]==1){
                // cout<<"hello"<<endl;
                dist[i][j]=1;
                q.push({i,j});
            }
        }
    }
    int count=0;
    while(!q.empty()){
        int x=q.front().first;
        int y=q.front().second;
        q.pop();
        if(x>=0 and y-1>=0 and x<n and y<n and dist[x][y-1]==INT_MAX){
            count++;
            dist[x][y-1]=1;
            q.push({x,y-1});
            // cout<<"hello"<<endl;
        }
        if(x>=0 and y+1>=0 and x<n and y<n and dist[x][y+1]==INT_MAX){
            count++;
             dist[x][y+1]=1;
            q.push({x,y+1});
            // cout<<"hello"<<endl;
        }
        if(x-1>=0 and y>=0 and x<n and y<n and dist[x-1][y]==INT_MAX){
            count++;
            dist[x-1][y]=1;
            q.push({x,y-1});
            // cout<<"hello"<<endl;
        }
        if(x+1>=0 and y>=0 and x<n and y<n and dist[x+1][y]==INT_MAX){
            count++;
            dist[x+1][y]=1;
            q.push({x,y-1});
            // cout<<"hello"<<endl;
        }
    }
    cout<<count<<endl;

    return 0;
}
